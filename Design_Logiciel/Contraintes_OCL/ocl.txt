1. Le port de départ et d'arrivée doit être le méme:

context Croisiere
   inv: self.detail.depart.uiniqueId = self.detail.arrivee.uniqueId


2. Un paquebot peut être assigné à plusieurs itinéraires tant qu'ils ne se chevauchent pas:

context Croisiere
   inv: Croisiere.allInstances() -> forAll(t1, t2: Croisiere 
           | t1 <> t2 AND t1.transport.id = t2.transport.id implies (t1.dateDepart > t2.dateArrivee OR t2.dateDepart > t1.dateArrivee))
